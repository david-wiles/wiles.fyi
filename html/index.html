func Render(w http.ResponseWriter, name string, funcs *template.FuncMap, data interface{}) {
    base := baseTemplates()
    if funcs != nil {
        base = base.Funcs(*funcs)
    }

    tmpl, err := base.ParseFiles("./templates" + name)
    if err != nil {
        Respond404(w, name)
        return
    }

    err = tmpl.ExecuteTemplate(w, strings.TrimPrefix(name, "/"), data)
    if err != nil {
        Respond404(w, name)
        return
    }
}
